% Definitions
% average value for population
\mu

% average value for sample
\overline{x}

%Bayes' Factor 1
% Bayes' Rule Symbols 0
p(b|a) =  \frac{p(a|b)  \times  p(a)}{p(b)}

% Bayes' Rule Symbols 1
p( \theta |x) =  \frac{p(x |  \theta )  \times  p( \theta )}{p(x)}

% Bayes' Rule Words 0
p(b|a) =  \frac{likelihood  \times  prior}{marginal\ likelihood}

% Bayes' Rule Words 1
p(hypothesis|data) =  \frac{p(data | hypothesis)  \times  p(hypothesis)}{p(data)}

% Coefficient of Determination (R-Squared)
r^{2} = 1.0 - \frac{sum \ of \ squared \ errors}{sum \ of \ squared \ variation \ from \ mean}
r^{2} = 1.0 - \frac{\sum_{i=1}^n  \big( y_{i} - \widehat{y}_i \big)^2}{\sum_{i=1}^n \big( y_{i} - \overline{y}_i \big)^2}

% Correlation
r = \frac{Covariance(x,y)}{(Standard Deviation(x))(Standard Deviation(y))}
r = \frac{Cov(x,y)}{s_xs_y}

% Linear Regression
\mid r\mid \ \geq \frac{2}{\sqrt{n}}

% Moments
\mu_{n}= \int_{-\infty}^\infty  \big(x - c\big)^nf \big(x\big)dx

% Population Covariance
\sigma_{xy} = \frac{\Sigma(x_i - \mu_x)(y_i - \mu_y)}{N}

% Posterior Ratio
R_{post} =  \frac{P(\theta_{x} | x)}{P(\theta_{y} | y)}

% Posterior Ratio Apply Bayes' Rules
R_{post} =  \frac{ \frac{P(x | \theta_{x})  \times P(  \theta_{x} ) }{P(x)} }{\frac{P(y | \theta_{y})  \times P(  \theta_{y} )}{P(y)}

% Posterior Ratio > 3
R_{post} > 3

% Posterior Ratio < 3
R_{post} < 3

% Probability of Alternate Event
P \big(E^c\big) = 1 - P \big(E\big)

% Probability of Combining Events
1 = P \big(s\big) = P \big(E  \cup E^c \big) = P \big(E \big) + P \big(E^c \big)

% Probability of Disjoint Events
P \big(E_1  \cup E_2  \cup ...  \cup  E_n \big) = P \big(E_1 \big) + P \big(E_2\big) + ... + P \big(E_n\big)

% Probability of Event
P \big(E\big) \geq 0

% Probability of Sample Space s
P \big(s\big) = 1

% Probability of Subset A in Set B
P \big(A \big)  \leq P \big(B \big)

% Probability of Union of Two Disjoint Sets
P \big(A \big) + P \big(B \big) = P \big( A \cap B \big)

% Sample Covariance
Cov(x,y) = S_{xy} = \frac{\Sigma(x_i - \overline{x})(y_i - \overline{y})}{n-1}

% Standard Deviation of x
s_x

% Weighted Average
\overline{x} = \frac{\sum_{i=1}^{n} \big(x_{i} \times w_{i} \big)}{\sum_{i=1}^{n} w_{i}}

% Variance: Population
\sigma^{2} =  \frac{ \sum  \big(X- \mu )^2  }{N}

% Variance: Sample
S^{2} =  \frac{ \sum  \big(X-M )^2  }{N-1}
