Input must be an \textarg{iterable}.
\begin{itemize}
  \item Returns True if:
    \begin{itemize}
      \item any of the elements of the iterable are true
    \end{itemize}
  \item Returns False if:
    \begin{itemize}
      \item the iterable is {\color{red}{\textbf{empty}}}
    \end{itemize}
\end{itemize}
